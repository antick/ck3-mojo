

namespace = chad_debug

chad_debug.0001 = { # assign someone as a garuda character event
	type = character_event
	title = chad_debug.0001.t 
	desc = chad_debug.0001.desc 

	theme = realm

	orphan = yes 

	left_portrait = {
		character = root 
		animation = happiness 
	}

	right_portrait = {
		character = scope:appointee
		animation = beg
	}

	trigger = {
		can_employ_court_position_type = garuda_court_position
		any_knight = {
			root = {
				can_appoint_char_to_court_position = {
					CHAR = prev
					COURT_POS = garuda_court_position
				}
			}
		}
	}

	immediate = {
		random_knight = {
			limit = {
				root = {
					can_appoint_char_to_court_position = {
						CHAR = prev
						COURT_POS = garuda_court_position
					}
				}
			}
			save_scope_as = appointee
		}
	}

	option = {
		name = chad_debug.0001.a 

		appoint_court_position = {
			recipient = scope:appointee
			court_position = garuda_court_position
		}
	}
}
