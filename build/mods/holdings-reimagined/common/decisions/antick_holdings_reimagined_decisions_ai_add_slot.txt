antick_holdings_reimagined_decision_ai_add_slot = {
	picture = "gfx/interface/illustrations/decisions/decision_major_religion.dds"

	ai_check_interval = 7

	is_shown = {
		is_ai = yes
		NOT = { has_game_rule = antick_holdings_reimagined_rule_ai_extra_slots_disabled }
		trigger_if = {
			limit = {
				is_landed = yes
			}
			any_directly_owned_province = {
				has_ongoing_construction = no
				has_free_building_slot = no
				antick_holdings_reimagined_trigger_ai_add_slot = yes
				antick_holdings_reimagined_value_ai_potential_slots > 0
			}
		}
		trigger_else = {
			is_theocratic_lessee = yes
			any_realm_province = {
				has_ongoing_construction = no
				has_holding_type = church_holding
				has_free_building_slot = no
				barony = {
					lessee = root
					is_leased_out = yes
					is_under_holy_order_lease = no
				}
				antick_holdings_reimagined_trigger_ai_add_slot = yes
				antick_holdings_reimagined_value_ai_potential_slots > 0
			}
		}
	}

	effect = {
		if = {
			limit = {
				is_landed = yes
			}
			random_directly_owned_province = {
				limit = {
					has_ongoing_construction = no
					has_free_building_slot = no
					antick_holdings_reimagined_trigger_ai_add_slot = yes
					antick_holdings_reimagined_value_ai_potential_slots > 0
				}
				save_temporary_scope_as = antick_holdings_reimagined_decision_scope_lotto
			}
		}
		else_if = {
			limit = {
				is_theocratic_lessee = yes
			}
			random_realm_province = {
				limit = {
					has_ongoing_construction = no
					has_holding_type = church_holding
					has_free_building_slot = no
					barony = {
						lessee = root
						is_leased_out = yes
						is_under_holy_order_lease = no
					}
					antick_holdings_reimagined_trigger_ai_add_slot = yes
					antick_holdings_reimagined_value_ai_potential_slots > 0
				}
				save_temporary_scope_as = antick_holdings_reimagined_decision_scope_lotto
			}
		}

		scope:antick_holdings_reimagined_decision_scope_lotto ?= {
			save_temporary_scope_value_as = {
				name = antick_holdings_reimagined_decision_scope_add_building_slot_cost_ai
				value = antick_holdings_reimagined_value_ai_slot_cost
			}
			save_temporary_scope_value_as = {
				name = antick_holdings_reimagined_decision_scope_add_building_slot_minimum_gold_ai
				value = antick_holdings_reimagined_value_ai_slot_minimum_cost
			}
			if = {
				limit = {
					root = {
						long_term_gold >= scope:antick_holdings_reimagined_decision_scope_add_building_slot_minimum_gold_ai
					}
				}
				add_province_modifier = extra_building_slot
				root = {
					remove_long_term_gold = scope:antick_holdings_reimagined_decision_scope_add_building_slot_cost_ai
				}
			}
		}
	}

	ai_will_do = {
		base = 100
	}
}
