antick_holdings_reimagined_consulate_can_construct_or_enabled_trigger = {
	trigger_if = {
		limit = {
			exists = scope:holder.capital_province
			this = scope:holder.capital_province
		}
	}
	trigger_else = {
		custom_description = {
			text = "antick_holdings_reimagined_must_be_your_capital_failed"
			always = no
		}
	}

	trigger_if = {
		limit = {
			scope:holder = {
				exists = dynasty
				dynasty = {
					or = {
						has_dynasty_perk = law_legacy_1
						has_dynasty_perk = erudition_legacy_1
					}
				}
			}
		}
	}
	trigger_else = {
		custom_description = {
			text = "antick_holdings_reimagined_need_law_1_or_erudite_1_failed"
			always = no
		}
	}
}

antick_holdings_reimagined_building_requirement_castle_city_church_tribe = {
	trigger_if = {
		limit = { has_holding_type = castle_holding }
		has_building_or_higher = castle_$LEVEL$
	}
	trigger_else_if = {
		limit = { has_holding_type = city_holding }
		has_building_or_higher = city_$LEVEL$
	}
	trigger_else_if = {
		limit = { has_holding_type = church_holding }
		has_building_or_higher = temple_$LEVEL$
	}
	trigger_else_if = {
		limit = { has_holding_type = tribal_holding }
		has_building_or_higher = tribe_$LEVEL$
	}
	trigger_else = {
		custom_description = {
			text = "building_requirement_castle_city_church_tribe_failed"
			always = no
		}
	}
}
