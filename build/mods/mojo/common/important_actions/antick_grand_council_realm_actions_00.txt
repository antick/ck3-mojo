action_empty_council_position = {
    type = alert

    check_create_action = {
        if = {
            limit = {
                highest_held_title_tier > tier_barony
                has_free_council_slot = yes

                OR = {
                    OR = {
                        AND = {
                            has_game_rule = antick_grand_council_player_enabled
                            highest_held_title_tier >= tier_empire
                            any_normal_councillor = {
                                count < 23
                            }
                        }

                        AND = {
                            has_game_rule = antick_grand_council_player_disabled
                            highest_held_title_tier >= tier_empire
                            any_normal_councillor = {
                                count < 6
                            }
                        }
                    }
                    OR = {
                        AND = {
                            has_game_rule = antick_grand_council_player_enabled
                            highest_held_title_tier >= tier_kingdom
                            any_normal_councillor = {
                                count < 17
                            }
                        }

                        AND = {
                            has_game_rule = antick_grand_council_player_disabled
                            highest_held_title_tier >= tier_kingdom
                            any_normal_councillor = {
                                count < 6
                            }
                        }
                    }
                    OR = {
                        AND = {
                            has_game_rule = antick_grand_council_player_enabled
                            highest_held_title_tier > tier_barony
                            any_normal_councillor = {
                                count < 11
                            }
                        }

                        AND = {
                            has_game_rule = antick_grand_council_player_disabled
                            highest_held_title_tier > tier_barony
                            any_normal_councillor = {
                                count < 6
                            }
                        }
                    }
                }
            }
            try_create_important_action = {
                important_action_type = action_empty_council_position
                actor = root
            }
        }
    }

    effect = {
        scope:actor = {
            open_view_data = {
                view = council_window
            }
        }
    }
}
